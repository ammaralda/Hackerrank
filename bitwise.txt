int bitwiseAnd(int N, int K) {

    int result;
    int max = 0;
    int border;
    
    
    for (int i = 0; i < N; i++){
        
        for (int j = i + 1; j <= N; j++){
            
            result = j & i;
           // cout << result << endl;
            if (result < K && max < result){
                max = result;
            }
        }
    }
    return max;
}
